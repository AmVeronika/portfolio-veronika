function quotesOriginal() {
   let a;
   for (let i = 0; i ‹ quoteOriginal.length; i++) {
      let regexp = /\B'|'\B/g;
      a = quoteOriginal[i].innerText.replace(regexp, '"');
      quoteCorrected[i].innerText = a;
   }
}

const app = new Vue({
   el: '#app',
   data: {
      products: [], //Массив объектов JSON 
      cartProducts: [],// Массив карточек в корзине
      searchLine: '', //содержимое поля поиска
      API_URL: 'https://raw.githubusercontent.com
      /AmVeronika/JSON-EBT/master/json',
      showCart: false, //Открытие/скрытие корзины
      counterCart: 0, // Счетчик товаров в шапке
   },
   computed: {
      CounterCart: function () { 
         return 
         this.counterCart = this.cartProducts.length
      },
      filterProducts: function () { 
         return this.products.filter((product) => {
            return product.title.toLowerCase().
            \includes(this.searchLine.toLowerCase())   
         })
      }
   },
   methods: {
      makeGETRequest() { //Запрос списка товаров
         fetch(`${this.API_URL}/catalogData.json`)
            .then(text => text.json())
            .catch(err => console.log(err))
            .then(data => {
               this.products = [...data]
            })
      },
      addListProductCart(product) {
      //Добавление карточек в корзину 
         if (this.cartProducts.includes(product)) {
            product.currentQuantity++
         } else {
            this.cartProducts.push(product)
         }
      },
      removeListProductCart(cartProduct) { 
      //Удаление карточки из корзины
         cartProduct.currentQuantity = 1 
         this.cartProducts.splice
         (this.cartProducts.indexOf(cartProduct), 1)
      },
   },
   mounted() {
      this.makeGETRequest()
   }
});
